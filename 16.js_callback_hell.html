콜백지옥

예시 요구사항

  • 만약 새로운 api요청을 통해 받은 응답을 활용해 새로운 요청을 보내야한다면, ajax 함수는 비동기로 동작하므로 반드시 success 안에서 동기적 호출을 진행해야하게 됨.
  • 이를 콜백지옥이라고하며, 가독성이 매우 떨어지는 문제점 발생
  • 개선 방법 : AJAX 기술 (axios, jquery)

jquery 라이브러리 사용 예시

const callbackHellDemo = () => {
    $(document).ready(() => {
    $.ajax({
        url: "https://jsonplaceholder.typicode.com/posts/1",
        type: "GET",
        dataType: "json",
        success: () => {
        console.log("B");
        $.ajax({
            url: "https://jsonplaceholder.typicode.com/posts/1",
            type: "GET",
            dataType: "json",
            success: () => {
            console.log("C");
            $.ajax({
                url: "https://jsonplaceholder.typicode.com/posts/1",
                type: "GET",
                dataType: "json",
                success: (result) => {
                console.log("D");
                console.log(result);
                }
            });
            }
        });
        }
    });
    });

    console.log("A");
};
